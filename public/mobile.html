Three calls, two languages.

The mobile SDK ships as a Swift package and an Android library. Same shape, same contract. Load. Personalize. Run.

Swift · iOS SwiftPM 
// Add: https://github.com/kolm-ai/kolm-swift
import Kolm

// 1. Load the signed artifact (App Group bundle).
let artifact = try await Kolm.load(
    bundle: "writing-style.kolm",
    publicKey: AppConfig.kolmPubKey
)

// 2. Personalize on the user's local data.
let session = try await artifact.personalize(
    with: UserCorpus(
        notes: notes,
        privacyScope: .strictlyOnDevice
    )
)

// 3. Run.
let reply = try await session.run(
    input: .text("Draft the Monday update")
)
// reply.text, reply.receipt, reply.latencyMs
Kotlin · Android Maven Central 
// Add: implementation("ai.kolm:runtime:0.1.0")
import ai.kolm.runtime.Kolm

// 1. Load the signed artifact.
val artifact = Kolm.load(
    asset = "writing-style.kolm",
    publicKey = BuildConfig.KOLM_PUB_KEY
)

// 2. Personalize on the user's local data.
val session = artifact.personalize(
    corpus = UserCorpus(
        notes = noteRepo.all(),
        privacyScope = PrivacyScope.STRICTLY_ON_DEVICE
    )
)

// 3. Run.
val reply = session.run(
    Input.Text("Draft the Monday update")
)
// reply.text, reply.receipt, reply.latencyMs
